can_be_warrior_trigger = {
	is_adult = yes
	age >= 16
	is_imprisoned = no
	prowess >= 7
	OR = {
		can_be_combatant_based_on_gender_trigger = { ARMY_OWNER = $ARMY_OWNER$ }
		AND = {
			is_female = yes
			OR = {
				religion = religion:germanic_religion
				religion = religion:slavic_religion
			}
			#Remove hashtags below if you only want norse faith rulers to be able to appoint shield-maidens.
			#liege = {
			#	OR = {
			#		religion = religion:germanic_religion
			#		religion = religion:slavic_religion
			#	}
			#}
			prowess >= 14
			OR = {
				has_trait = education_martial_1
				has_trait = education_martial_2
				has_trait = education_martial_3
				has_trait = education_martial_4
				has_trait = blademaster_2
				has_trait = blademaster_3
				has_trait = hunter_2
				has_trait = hunter_3
			}
		}
	}
	NOT = {
		has_trait = incapable
		has_trait = pregnant
	}
}
