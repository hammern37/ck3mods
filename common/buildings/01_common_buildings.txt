horse_01 = {
	construction_time = slow_construction_time

	can_construct_potential = {
		building_horse_requirement_terrain = yes
		building_requirement_castle_city_church = { LEVEL = 01 }
		OR = {
			scope:holder = {
				faith = { religion_tag = christianity_religion }
		}
			scope:holder = {
				faith = { religion_tag = shamanism_regligion }
			}
		}
	}

	can_construct_showing_failures_only = {
		building_requirement_tribal = no
	}
	
	cost_gold = 150
	
	province_modifier = {
		monthly_income = 0.2
	}

	character_modifier = {
		light_cavalry_damage_add = 1
		light_cavalry_pursuit_add = 1
	}
	
	next_building = horse_02
	
	type_icon = "icon_building_horse.dds"
	
	ai_value = {
		base = 1
		modifier = {
			factor = 0
			building_regimental_grounds_requirement_terrain = yes
			NOT = {
				has_building_or_higher = regimental_grounds_01
			}
		}
	}
}
