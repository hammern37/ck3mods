quarterly_playable_pulse = {
	# checks for court physician 4 times per year
	effect = {
		if = {
			limit = {
				NOT = {
					any_relation = {
						type = court_physician
					}
				}
			}
			add_character_flag = {
				flag = flag_no_physician
			}
		}
		else_if = {
			limit = {
				any_relation = {
					type = court_physician
				}
			}
			remove_character_flag = flag_no_physician
		}
	}
}
