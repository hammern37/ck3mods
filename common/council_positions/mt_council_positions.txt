councillor_master_of_the_horses = {
	name = councillor_master_of_the_horses
	skill = martial
	tooltip = game_concept_master_of_the_horses_desc
	valid_position = {
		highest_held_title_tier >= tier_duchy
	}
	valid_character = {
		can_be_master_of_the_horses_trigger = yes
	}

	modifier = {
		name = councillor_master_of_the_horses_modifier
		prowess = 1
		monthly_prestige = 0.5
		monthly_martial_lifestyle_xp_gain_mult = 0.01
	}	

	on_get_position = {
		gave_minor_title_effect = yes 					
	}
	
	on_fired_from_position = {
	if = {
		limit = { is_powerful_vassal = scope:liege }
		remove_opinion = 
			{
			modifier = gave_minor_title_normal_vassal_opinion
			target = liege
			}
		}
	else if = {
		remove_opinion = 
			{
			modifier = gave_minor_title_powerful_vassal_opinion
			target = liege
			}
		}
	}		
	portrait_animation = master_of_the_horses
}
