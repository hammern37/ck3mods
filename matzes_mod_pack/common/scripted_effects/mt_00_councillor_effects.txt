gave_minor_title_effect =
{
	if = {
		limit = { is_powerful_vassal = scope:liege }
		add_opinion = 
			{
			modifier = gave_minor_title_normal_vassal_opinion
			target = liege
			}
		}
	else if = {
			add_opinion = 
			{
			modifier = gave_minor_title_powerful_vassal_opinion
			target = liege
			}
		}
}
