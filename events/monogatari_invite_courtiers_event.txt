
namespace = monogatari_invite_courtiers

# Invite male courtier
monogatari_invite_courtiers.0001 = {
	type = character_event
	title = monogatari_invite_courtiers.0001.title
	desc = monogatari_invite_courtiers.0001.desc
	
	theme = realm
	
	left_portrait = {
		character = root
		animation = idle
	}
	right_portrait = {
		character = scope:target
		animation = idle
	}

	trigger = { exists = capital_province } #If I lost it before I got this event, I shouldn't get a physician

	immediate = {
		# Create random courtier
		create_character = {
			employer = root
			age = { 16 50 }
			gender = male
			#gender_female_chance = {
			#	value = 50
			#	if = {
			#		limit = { root = { is_attracted_to_men = yes } }
			#		add = -50
			#	}
			#	if = {
			#		limit = { root = { is_attracted_to_women = yes } }
			#		add = 50
			#	}
			#}
			random_traits = yes
			faith = root.faith
			culture = root.culture
			dynasty = generate
			save_scope_as = target
		}
	}

	# Hire them
	option = {
		name = monogatari_invite_courtiers.0001.a
	}

	# Don't hire them
	option = {
		name = monogatari_invite_courtiers.0001.b
		hidden_effect = {
			scope:target = {		
				death = {
					death_reason = death_disappearance
				}
			}
		}
	}
}
